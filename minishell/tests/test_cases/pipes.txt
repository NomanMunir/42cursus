# Pipe Operations Test Cases
# Test various pipe operations

# Simple pipes
echo hello | cat
echo "hello world" | wc -w
ls /bin | head -5
cat /etc/passwd | grep root

# Multiple pipes
echo "one two three four five" | tr ' ' '\n' | wc -l
ls -1 /bin | head -10 | tail -5
echo "test line" | cat | cat | cat

# Pipes with built-ins
export TEST="pipe test"
echo $TEST | cat
pwd | cat
env | grep PATH | head -1

# Complex pipes
echo -e "line1\nline2\nline3" | grep "line2"
echo "HELLO WORLD" | tr 'A-Z' 'a-z'
seq 1 10 | grep "5"

exit
